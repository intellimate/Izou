003/**
004 * FileSubscribers are usually paired with {@link ReloadableFile} objects in the {@link FilePublisher}, where their
005 * update method is triggered with a file change pertaining to the reloadable file.
006 */
007public interface FileSubscriber {
008    /**
009     * Method that is called when a file pertaining to a reloadableFile is changed.
010     */
011    void update();
012}
