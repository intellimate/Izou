001package org.intellimate.izou.system;
002
003import org.intellimate.izou.addon.AddOnModel;
004import org.apache.logging.log4j.spi.ExtendedLogger;
005import org.intellimate.izou.system.context.*;
006import org.intellimate.izou.system.context.System;
007
008/**
009 * @author Leander Kurscheidt
010 * @version 1.0
011 */
012public interface Context {
013    /**
014     * returns the API used for interaction with Events
015     * @return Events
016     */
017    Events getEvents();
018
019    /**
020     * returns the API used for interaction with Resource
021     * @return Resource
022     */
023    Resources getResources();
024
025    /**
026     * returns the API used for interaction with Files
027     * @return Files
028     */
029    Files getFiles();
030
031    /**
032     * returns the API used to log
033     * @return Logger
034     */
035    ExtendedLogger getLogger();
036
037    /**
038     * returns the API used to manage the ThreadPool
039     * @return ThreadPool
040     */
041    ThreadPool getThreadPool();
042
043    /**
044     * returns the API to manage the Activators
045     * @return Activator
046     */
047    Activators getActivators();
048
049    /**
050     * returns the API used to manage the OutputPlugins and OutputExtensions
051     * @return Output
052     */
053    Output getOutput();
054
055    /**
056     * retruns the API used to interact with Izou.
057     * @return System.
058     */
059    System getSystem();
060
061    /**
062     * gets addOn
063     *
064     * @return the addOn
065     */
066    AddOnModel getAddOn();
067}
