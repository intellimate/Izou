027/**
028 * OutputManager manages all output plugins and is the main class anyone outside the output package should talk to.
029 * It can register/remove new output-plugins and add/delete output-extensions
030 */
031public class OutputManager extends IzouModule implements AddonThreadPoolUser {
032
033    /**
034     * a list that contains all the registered output-plugins of Jarvis
035     */
036    private IdentifiableSet<OutputPluginModel<?, ?>> outputPlugins;
037
038    /**
039     * HashMap that stores OutputPlugins and the Future objects representing the Task
040     */
041    private HashMap<String, Future> futureHashMap;
042
043    /**
044     * this HashMap stores all added OutputExtensions
045     */
046    private HashMap<String, IdentifiableSet<OutputExtensionModel<?, ?>>> outputExtensions;
047
048
049    /**
050     * Creates a new output-manager with a list of output-plugins
051     * @param main the main instance started from
052     */
053    public OutputManager(Main main) {
054        super(main);
055        outputPlugins = new IdentifiableSet<>();
056        futureHashMap = new HashMap<>();
057        outputExtensions = new HashMap<>();
058    }
059
060    /**
061     * adds outputPlugin to outputPluginList, starts a new thread for the outputPlugin, and stores the future object in a HashMap
062     * @param outputPlugin OutputPlugin to add
063     * @throws IllegalIDException not yet implemented
064     */
065    public void addOutputPlugin(OutputPluginModel<?, ?> outputPlugin) throws IllegalIDException {
066        if (!futureHashMap.containsKey(outputPlugin.getID())) {
067            outputPlugins.add(outputPlugin);
068            futureHashMap.put(outputPlugin.getID(), submit(outputPlugin));
069        } else {
070            if (futureHashMap.get(outputPlugin.getID()).isDone()) {
071                futureHashMap.remove(outputPlugin.getID());
072                futureHashMap.put(outputPlugin.getID(), submit(outputPlugin));
073            }
074        }
075    }
076
077    /**
078     * removes the OutputPlugin and stops the thread
079     * @param outputPlugin the outputPlugin to remove
080     */
081    public void removeOutputPlugin(OutputPluginModel outputPlugin) {
082        Future future = futureHashMap.remove(outputPlugin.getID());
083        if (future != null) {
084            future.cancel(true);
085        }
086        outputPlugins.remove(outputPlugin);
087    }
088
089    /**
090     * adds output extension to desired outputPlugin
091     *
092     * adds output extension to desired outputPlugin, so that the output-plugin can start and stop the outputExtension
093     * task as needed. The outputExtension is specific to the output-plugin
094     *
095     * @param outputExtension the outputExtension to be added
096     * @throws IllegalIDException not yet implemented
097     */
098    public void addOutputExtension(OutputExtensionModel<?, ?> outputExtension) throws IllegalIDException {
099        if(outputExtensions.containsKey(outputExtension.getPluginId())) {
100            outputExtensions.get(outputExtension.getPluginId()).add(outputExtension);
101        }
102        else {
103            IdentifiableSet<OutputExtensionModel<?, ?>> outputExtensionList = new IdentifiableSet<>();
104            outputExtensionList.add(outputExtension);
105            outputExtensions.put(outputExtension.getPluginId(), outputExtensionList);
106        }
107        IdentificationManager.getInstance().getIdentification(outputExtension)
108                .ifPresent(id -> outputPlugins.stream()
109                        .filter(outputPlugin -> outputPlugin.getID().equals(outputExtension.getPluginId()))
110                        .forEach(outputPlugin -> outputPlugin.outputExtensionAdded(id)));
111    }
112
113    /**
114     * removes the output-extension of id: extensionId from outputPluginList
115     *
116     * @param outputExtension the OutputExtension to remove
117     */
118    public void removeOutputExtension(OutputExtensionModel<?, ?> outputExtension) {
119        IdentifiableSet<OutputExtensionModel<?, ?>> outputExtensions =
120                this.outputExtensions.get(outputExtension.getPluginId());
121        if (outputExtensions != null)
122            outputExtensions.remove(outputExtension);
123        IdentificationManager.getInstance().getIdentification(outputExtension)
124                .ifPresent(id -> outputPlugins.stream()
125                        .filter(outputPlugin -> outputPlugin.getID().equals(outputExtension.getPluginId()))
126                        .forEach(outputPlugin -> outputPlugin.outputExtensionRemoved(id)));
127    }
128
129    /**
130     * gets the Event and sends it to the right outputPlugin for further processing
131     *
132     * passDataToOutputPlugins is the main method of OutputManger. It is called whenever the output process has to be started
133     *
134     * @param event an Instance of Event
135     */
136    public void passDataToOutputPlugins(EventModel event) {
137        IdentificationManagerM identificationManager = IdentificationManager.getInstance();
138        List<Identification> allIds = outputPlugins.stream()
139                .map(identificationManager::getIdentification)
140                .filter(Optional::isPresent)
141                .map(Optional::get)
142                .collect(Collectors.toList());
143        
144        HashMap<Integer, List<Identification>> outputPluginBehaviour = event.getEventBehaviourController()
145                .getOutputPluginBehaviour(allIds);
146
147        @SuppressWarnings("unchecked")
148        Set<OutputPluginModel> outputPluginsCopy = (Set<OutputPluginModel>) this.outputPlugins.clone();
149        
150        Function<List<Identification>, List<OutputPluginModel>> getOutputPlugin = ids -> ids.stream()
151                .map(id -> outputPluginsCopy.stream()
152                        .filter(outputPlugin -> outputPlugin.isOwner(id))
153                        .findFirst()
154                        .orElseGet(null))
155                .filter(Objects::nonNull)
156                .peek(outputPluginsCopy::remove)
157                .collect(Collectors.toList());
158        
159        outputPluginBehaviour.entrySet().stream()
160                .sorted(Comparator.comparingInt((Map.Entry<Integer, List<Identification>> x) -> x.getKey()).reversed())
161                .flatMap(entry -> getOutputPlugin.apply(entry.getValue()).stream())
162                .distinct()
163                .forEach(op -> processOutputPlugin(event, op));
164        
165        outputPluginsCopy.forEach(op -> processOutputPlugin(event, op));
166    }
167
168    private void processOutputPlugin(EventModel event, OutputPluginModel outputPlugin) {
169        debug("processing outputPlugin: " + outputPlugin.getID() + " for event: " + event.getDescriptors().toString());
170       final Lock lock = new ReentrantLock();
171       final Condition processing = lock.newCondition();
172
173       Consumer<Boolean> consumer = noParam -> {
174           lock.lock();
175           processing.signal();
176           lock.unlock();
177       };
178        
179        ResourceMinimalImpl<Consumer<Boolean>> resource = IdentificationManager.getInstance().getIdentification(this)
180               .map(id -> new ResourceMinimalImpl<>(outputPlugin.getID(), id, consumer, null))
181               .orElse(new ResourceMinimalImpl<>(outputPlugin.getID(), null, consumer, null));
182       event.getListResourceContainer().addResource(resource);
183
184       outputPlugin.addToEventList(event);
185
186       boolean finished = false;
187       try {
188           lock.lock();
189           finished = processing.await(100, TimeUnit.SECONDS);
190       } catch (InterruptedException e) {
191           error("Waiting for OutputPlugins interrupted", e);
192       } finally {
193           lock.unlock();
194       }
195       if(finished) {
196           debug("OutputPlugin: " + outputPlugin.getID() + " finished");
197       } else {
198           error("OutputPlugin: " + outputPlugin.getID() + " timed out");
199       }
200   }

202    /**
203     * returns all the associated OutputExtensions
204     * @param outputPlugin the OutputPlugin to search for
205     * @return a List of Identifications
206     */
207    //TODO: TEST!!!!
208    public List<Identification> getAssociatedOutputExtension(OutputPluginModel<?, ?> outputPlugin) {
209        IdentifiableSet<OutputExtensionModel<?, ?>> outputExtensions = this.outputExtensions.get(outputPlugin.getID());
210        IdentificationManagerM identificationManager = IdentificationManager.getInstance();
211        return filterType(outputExtensions, outputPlugin).stream()
212                .map(identificationManager::getIdentification)
213                .filter(Optional::isPresent)
214                .map(Optional::get)
215                .collect(Collectors.toList());
216    }

218    /**
219     * checks for the right type
220     * @param outputExtensions the OutputExtensions to check
221     * @param outputPlugin the OutputPlugin to check against
222     * @return a List of filtered OutputExtensions
223     */
224    //TODO: TEST!!!!
225    @SuppressWarnings("SimplifiableIfStatement")
226    private List<OutputExtensionModel<?, ?>> filterType(Collection<OutputExtensionModel<?, ?>> outputExtensions,
227                                                   OutputPluginModel<?, ?> outputPlugin) {
228        BiPredicate<TypeToken<?>, TypeToken<?>> isAssignable = (first, second) -> {
229            if (first == null) {
230                return second == null;
231            } else if (second != null) {
232                return first.isAssignableFrom(second);
233            } else {
234                return false;
235            }
236        };
237        return outputExtensions.stream()
238                .filter(outputExtension ->
239                        isAssignable.test(outputExtension.getArgumentType(), outputPlugin.getArgumentType()))
240                .filter(outputExtension ->
241                        isAssignable.test(outputExtension.getReturnType(), outputPlugin.getReceivingType()))
242                .collect(Collectors.toList());
243    }

245    /**
246     * starts every associated OutputExtension
247     * @param outputPlugin the OutputPlugin to generate the Data for
248     * @param t the argument or null
249     * @param event the Event to generate for
250     * @param <T> the type of the argument
251     * @param <X> the return type
252     * @return a List of Future-Objects
253     */
254    //TODO: Test! is it working?
255    public <T, X> List<CompletableFuture<X>> generateAllOutputExtensions(OutputPluginModel<T, X> outputPlugin,
256                                                                                T t, EventModel event) {
257        IdentifiableSet<OutputExtensionModel<?, ?>> extensions = outputExtensions.get(outputPlugin.getID());
258        if (extensions == null)
259            return new ArrayList<>();
260        return filterType(extensions, outputPlugin).stream()
261                .map(extension -> {
262                    try {
263                        //noinspection unchecked
264                        return (OutputExtensionModel<X, T>) extension;
265                    } catch (ClassCastException e) {
266                        return null;
267                    }
268                })
269                .filter(Objects::nonNull)
270                .filter(outputExtension -> outputExtension.canRun(event))
271                .map(extension -> submit(() -> extension.generate(event, t)))
272                .collect(Collectors.toList());
273    }
