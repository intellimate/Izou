007import java.io.File;
008import java.io.FilePermission;

016/**
017 * The FilePermissionModule is used to check the access to files
018 * @author LeanderK
019 * @version 1.0
020 */
021public class FilePermissionModule extends PermissionModule {
022    private final List<String> allowedReadDirectories;
023    private final List<String> allowedReadFiles;
024    private final List<File> allowedWriteDirectories;
025    private final List<File> forbiddenWriteDirectories;
026    private final List<String> forbiddenWriteFilesNames;
027    private final String allowedReadFileTypesRegex;
028    private final String allowedWriteFileTypesRegex;
029
030    /**
031     * Creates a new PermissionModule
032     *
033     * @param main the instance of main
034     */
035    FilePermissionModule(Main main, SecurityManager securityManager) {
036        super(main, securityManager);
037        allowedReadDirectories = new ArrayList<>();
038        allowedReadFiles = new ArrayList<>();
039        forbiddenWriteDirectories = new ArrayList<>();
040        forbiddenWriteFilesNames = new ArrayList<>();
041        forbiddenWriteFilesNames.add("addon_config.properties");
042        allowedReadFiles.add("/dev/random");
043        allowedReadFiles.add("/dev/urandom");
044        allowedWriteDirectories = new ArrayList<>();
045        allowedReadFileTypesRegex = "(txt|properties|xml|class|json|zip|ds_store|mf|jar|idx|log|dylib|mp3|dylib|certs|"
046                + "so)";
047        allowedWriteFileTypesRegex = "(txt|properties|xml|json|idx|log)";
048        allowedReadDirectories.addAll(Arrays.asList(System.getProperty("java.ext.dirs").split(":")));
049        allowedReadDirectories.add(System.getProperty("java.home"));
050        allowedReadDirectories.add(main.getFileSystemManager().getLogsLocation().getAbsolutePath());
051        allowedReadDirectories.add(main.getFileSystemManager().getPropertiesLocation().getAbsolutePath());
052        allowedReadDirectories.add(main.getFileSystemManager().getResourceLocation().getAbsolutePath());
053        allowedReadDirectories.add(main.getFileSystemManager().getIzouJarLocation().getAbsolutePath());
054        allowedReadDirectories.add(main.getFileSystemManager().getLibLocation().getAbsolutePath());
055        allowedWriteDirectories.add(main.getFileSystemManager().getResourceLocation());
056        allowedWriteDirectories.add(main.getFileSystemManager().getLogsLocation());
057        allowedWriteDirectories.add(main.getFileSystemManager().getPropertiesLocation());
058        allowedWriteDirectories.add(main.getFileSystemManager().getResourceLocation());
059        allowedWriteDirectories.add(main.getFileSystemManager().getLibLocation());
060        if (getMain().getFileSystemManager().getIzouJarLocation().isDirectory()) {
061            allowedWriteDirectories.add(new File(getMain().getFileSystemManager().getIzouJarLocation() + File.separator + "META-INF" + File.separator + "services"));
062        } else {
063            URL metaInfo = this.getClass().getClassLoader().getResource("META-INF/services");
064            if (metaInfo != null)
065                allowedWriteDirectories.add(new File(metaInfo.getFile()));
066        }
067
068        if (Boolean.getBoolean("debug")) {
069            allowedReadDirectories.add(System.getProperty("user.home") + File.separator + ".m2");
070            allowedWriteDirectories.add(new File(System.getProperty("user.home") + File.separator + ".m2"));
071            allowedWriteDirectories.add(main.getFileSystemManager().getIzouJarLocation());
072            allowedWriteDirectories.add(new File(System.getProperty("java.home")));
073        }
074    }
075
076    /**
077     * returns true if able to check permissions
078     *
079     * @param permission the permission to check
080     * @return true if able to, false if not
081     */
082    @Override
083    public boolean canCheckPermission(Permission permission) {
084        return permission instanceof FilePermission;
085    }
086
087    /**
088     * Checks if the given addOn is allowed to access the requested service and registers them if not yet registered.
089     *
090     * @param permission the Permission to check
091     * @param addon      the identifiable to check
092     * @throws IzouPermissionException thrown if the addOn is not allowed to access its requested service
093     */
094    @Override
095    public void checkPermission(Permission permission, AddOnModel addon) throws IzouPermissionException {
096        String canonicalName = permission.getName().intern().toLowerCase();
097        String canonicalAction =  permission.getActions().intern().toLowerCase();
098
099        if (canonicalName.contains("all files") || canonicalAction.equals("execute")) {
100            throw getException(permission.getName());
101        }
102        if (canonicalAction.equals("read")) {
103            fileReadCheck(canonicalName);
104        }
105        fileWriteCheck(canonicalName, addon);
106    }
107
108    /**
109     * Determines if the file at the given file path is safe to read from in all aspects, if so returns true, else false
110     *
111     * @param filePath the path to the file to read from
112     */
113    void fileReadCheck(String filePath) {
114        File potentialFile = new File(filePath);
115        String canonicalPath;
116        try {
117            canonicalPath = potentialFile.getCanonicalPath();
118        } catch (IOException e) {
119            error("Error getting canonical path", e);
120            throw getException(filePath);
121        }
122
123        for (String file : allowedReadFiles) {
124            if (canonicalPath.contains(file)) {
125                return;
126            }
127        }
128
129        boolean allowedDirectory = false;
130        for (String dir : allowedReadDirectories) {
131            if (canonicalPath.contains(dir)) {
132                allowedDirectory = true;
133                break;
134            }
135        }
136        if (!allowedDirectory) {
137            throw getException(filePath);
138        }
139        /*
140        String[] pathParts = canonicalPath.split(File.separator);
141        String lastPathPart = pathParts[pathParts.length - 1].toLowerCase();
142
143        String[] pathPeriodParts = lastPathPart.split("\\.");
144        String fileExtension = pathPeriodParts[pathPeriodParts.length - 1].toLowerCase();
145
146        if (!getSecurityManager().getSecureAccess().checkForExistingFileOrDirectory(canonicalPath)
147                || getSecurityManager().getSecureAccess().checkForDirectory(canonicalPath)) {
148            return;
149        }
150
151        Pattern pattern = Pattern.compile(allowedReadFileTypesRegex);
152        Matcher matcher = pattern.matcher(fileExtension);
153        if (!matcher.matches() || fileExtension.equals(lastPathPart))
154            throw getException(filePath);*/
155    }
156
157    /**
158     * Determines if the file at the given file path is safe to write to in all aspects, if so returns true, else false
159     *
160     * @param filePath the path to the file to write to
161     * @param addOnModel the AddonModel
162     */
163    void fileWriteCheck(String filePath, AddOnModel addOnModel) {
164        File request;
165        try {
166            request =  new File(filePath).getCanonicalFile();
167        } catch (IOException e) {
168            error("Error getting canonical path", e);
169            throw getException(filePath);
170        }
171
172        isForbidden(request, addOnModel);
173
174        boolean success = false;
175        if (allowedWriteDirectories.stream()
176                .anyMatch(compare -> request.toPath().startsWith(compare.toPath()))) {
177            success = true;
178        }
179
180        for (String name : forbiddenWriteFilesNames) {
181            if (request.getName().equals(name)) {
182                success = false;
183            }
184        }
185
186        if (!success) {
187            throw getException(filePath);
188        }
189
190        if (!getSecurityManager().getSecureAccess().checkForExistingFileOrDirectory(request.toString())
191                || getSecurityManager().getSecureAccess().checkForDirectory(request.toString())) {
192            return;
193        }
194        /*
195        Pattern pattern = Pattern.compile(allowedWriteFileTypesRegex);
196        Matcher matcher = pattern.matcher(Files.getFileExtension(request.toString()));
197        if (!matcher.matches())
198            throw getException(filePath);*/
199    }
200
201    /**
202     * throws an Exception if the
203     * @param request the requested File
204     * @param addOnModel the AddonModel
205     */
206    private void isForbidden(File request, AddOnModel addOnModel) {
207        if (forbiddenWriteDirectories.stream()
208                .anyMatch(compare -> request.toPath().startsWith(compare.toPath()))) {
209            throw getException("file: " + request.toString() + " is forbidden. Attempt made by: "
210                    + addOnModel.getID());
211        }
212    }
