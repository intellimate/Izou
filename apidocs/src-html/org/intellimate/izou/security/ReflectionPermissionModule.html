001package org.intellimate.izou.security;
002
003import org.intellimate.izou.addon.AddOnManager;
004import org.intellimate.izou.addon.AddOnModel;
005import org.intellimate.izou.main.Main;
006
007import java.lang.reflect.ReflectPermission;
008import java.security.Permission;
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * The
014 */
015public class ReflectionPermissionModule extends PermissionModule {
016    private List<Class> forbiddenReflections;
017
018    public ReflectionPermissionModule(Main main, SecurityManager securityManager) {
019        super(main, securityManager);
020        // Security package
021        forbiddenReflections = new ArrayList<>();
022        forbiddenReflections.add(AudioPermissionModule.class);
023        forbiddenReflections.add(FilePermissionModule.class);
024        forbiddenReflections.add(PermissionManager.class);
025        forbiddenReflections.add(PermissionModule.class);
026        forbiddenReflections.add(ReflectionPermissionModule.class);
027        forbiddenReflections.add(RootPermission.class);
028        forbiddenReflections.add(SecureAccess.class);
029        forbiddenReflections.add(SecurityBreachHandler.class);
030        forbiddenReflections.add(SecurityManager.class);
031        forbiddenReflections.add(SecurityFunctions.class);
032        forbiddenReflections.add(SocketPermissionModule.class);
033
034        // Other
035        forbiddenReflections.add(AddOnManager.class);
036        forbiddenReflections.add(Main.class);
037    }
038
039    @Override
040    public boolean canCheckPermission(Permission permission) {
041        return permission instanceof ReflectPermission && "suppressAccessChecks".equals(permission.getName());
042    }
043
044    @Override
045    public void checkPermission(Permission permission, AddOnModel addon) throws SecurityException {
046        for (StackTraceElement elem : Thread.currentThread().getStackTrace()) {
047            if ("Test".equals(elem.getClassName()) && "badSetAccessible".equals(elem.getMethodName())) {
048                throw getException("Reflection not allowed here.");
049            }
050        }
051    }
052}
