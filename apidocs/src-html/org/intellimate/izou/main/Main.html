079     */
080    public Main(List<AddOnModel> addOns, boolean disableLibFolder) {
081        this(addOns, false, disableLibFolder);
082    }
083
084    /**
085     * If you want to debug your Plugin, you can get an Main instance with this Method
086     *
087     * @param javaFX true if javaFX should be started, false otherwise
088     * @param disableLibFolder  if true, izou will not load plugin from the lib-folder
089     * @param addOns a List of AddOns to run
090     */
091    public Main(List<AddOnModel> addOns, boolean javaFX, boolean disableLibFolder) {
092        fileLogger.debug("Starting Izou");
093        fileLogger.debug("Initializing...");
094        systemInitializer = initSystem();
095        if (addOns != null) {
096            fileLogger.debug("setting to debug");
097            System.setProperty("debug", "true");
098        }
099
100        internalIdentificationManager = new InternalIdentificationManager(this);
101        addOnManager = new AddOnManager(this);
102        threadPoolManager = new ThreadPoolManager(this);
103        izouLogger = new IzouLogger();
104        outputManager = new OutputManager(this);
105        resourceManager = new ResourceManager(this);
106        eventDistributor = new EventDistributor(this);
107        localEventManager = new LocalEventManager(this);
108        threadPoolManager.getIzouThreadPool().submit(localEventManager);
109        activatorManager = new ActivatorManager(this);
110        filePublisher = new FilePublisher(this);
111        soundManager = new SoundManager(this);
112
113        fileManager = initFileManager();
114
115        fileLogger.debug("Done initializing.");
116        fileLogger.debug("Adding addons..");
117
118        // Starting security manager
119        systemMail = initMail();
120        securityManager = startSecurity(systemMail);
121
122        if (addOns != null && !disableLibFolder) {
123            fileLogger.debug("adding addons from the parameter without registering");
124            addOnManager.addAddOnsWithoutRegistering(addOns);
125        } else if (addOns != null) {
126            fileLogger.debug("adding and registering addons from the parameter");
127            addOnManager.addAndRegisterAddOns(addOns);
128        }
129        if (!disableLibFolder) {
130            fileLogger.debug("retrieving addons & registering them");
131            addOnManager.retrieveAndRegisterAddOns();
132        }
133    }

143    private SystemMail initMail() {
144        // Create system mail
145        SystemMail mailTemp = null;
146        try {
147            mailTemp = SystemMail.createSystemMail();
148        } catch (IllegalAccessException e) {
149            fileLogger.fatal("Unable to create a system mail object");
150        }
151        return mailTemp;
152    }
153
154    private void setUpJavaFX(boolean javaFX) {
155        // Starts javaFX if desired
156        if (javaFX) {
157            jfxToolKitInit = new AtomicBoolean(false);
158            JavaFXInitializer.initToolKit();
159            fileLogger.debug("Initializing JavaFX ToolKit");
160
161            long startTime = System.currentTimeMillis();
162            long duration = 0;
163            while (!jfxToolKitInit.get() && duration < INIT_TIME_LIMIT) {
164                duration = System.currentTimeMillis() - startTime;
165                try {
166                    Thread.sleep(5000);
167                } catch (InterruptedException e) {
168                    fileLogger.error("Error happened while thread was sleeping", e);
169                }
170            }
171
172            if (!jfxToolKitInit.get()) {
173                fileLogger.error("Unable to Initialize JavaFX ToolKit");
174            }
175            fileLogger.debug("Done initializing JavaFX ToolKit");
176        }
177    }
178
179    private SecurityManager startSecurity(SystemMail systemMail) {
180        SecurityManager securityManagerTemp;
181        try {
182            securityManagerTemp = SecurityManager.createSecurityManager(systemMail, this);
183        } catch (IllegalAccessException e) {
184            securityManagerTemp = null;
185            fileLogger.fatal("Security manager already exists", e);
186        }
187        try {
188            System.setSecurityManager(securityManagerTemp);
189        } catch (SecurityException e) {
190            fileLogger.fatal("Security manager already exists", e);
191        }
192        return securityManagerTemp;
193    }
194
195    private FileManager initFileManager() {
196        FileManager fileManagerTemp;
197        try {
198            fileManagerTemp = new FileManager(this);
199        } catch (IOException e) {
200            fileManagerTemp = null;
201            fileLogger.fatal("Failed to create the FileManager", e);
202        }
203
204        return fileManagerTemp;
205    }

263    private SystemInitializer initSystem() {
264        // Setting up file system
265        this.fileSystemManager = new FileSystemManager(this);
266        try {
267            fileSystemManager.createIzouFileSystem();
268        } catch (IOException e) {
269            fileLogger.fatal("Failed to create the FileSystemManager", e);
270        }
271
272        SystemInitializer systemInitializer = new SystemInitializer();
273        systemInitializer.initSystem();
274        return systemInitializer;
