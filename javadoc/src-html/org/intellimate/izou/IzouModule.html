001package org.intellimate.izou;
002
003import org.apache.logging.log4j.Level;
004import org.apache.logging.log4j.LogManager;
005import org.apache.logging.log4j.spi.AbstractLogger;
006import org.intellimate.izou.identification.Identifiable;
007import org.intellimate.izou.identification.IdentificationManager;
008import org.intellimate.izou.main.Main;
009
010/**
011 * The base class for each IzouModule
012 * FIXME: What is an izou module exactly?
013 *
014 * @author Leander Kurscheidt
015 * @version 1.0
016 */
017public abstract class IzouModule implements MainProvider, Identifiable {
018    private static final String FQCN = IzouModule.class.getName();
019    private String ID = this.getClass().getCanonicalName();
020    protected Main main;
021    protected final AbstractLogger log = (AbstractLogger) LogManager.getLogger(this.getClass());
022
023    public IzouModule(Main main) {
024        this.main = main;
025        if (!IdentificationManager.getInstance().registerIdentification(this)) {
026            log.fatal("unable to register! " + getClass().getCanonicalName());
027        }
028    }
029
030    /**
031     * returns the instance of Main
032     *
033     * @return Main
034     */
035    @Override
036    public Main getMain() {
037        return main;
038    }
039
040    /**
041     * Used to log messages at debug level
042     *
043     * @param msg the message
044     * @param e   the Throwable
045     */
046    @Override
047    public void debug(String msg, Throwable e) {
048        log.logIfEnabled(FQCN, Level.DEBUG, null, msg, e);
049    }
050
051    /**
052     * Used to log messages at debug level
053     *
054     * @param msg the message
055     */
056    @Override
057    public void debug(String msg) {
058        log.logIfEnabled(FQCN, Level.DEBUG, null, msg, (Object) null);
059    }
060
061    /**
062     * Used to log messages at error level
063     *
064     * @param msg the message
065     * @param e   the Throwable
066     */
067    @Override
068    public void error(String msg, Throwable e) {
069        log.logIfEnabled(FQCN, Level.ERROR, null, msg, e);
070    }
071
072    /**
073     * Used to log messages at error level
074     *
075     * @param msg the message
076     */
077    @Override
078    public void error(String msg) {
079        log.logIfEnabled(FQCN, Level.ERROR, null, msg, (Object) null);
080    }
081
082    /**
083     * An ID must always be unique.
084     * A Class like Activator or OutputPlugin can just provide their .class.getCanonicalName()
085     * If you have to implement this interface multiple times, just concatenate unique Strings to
086     * .class.getCanonicalName()
087     *
088     * @return A String containing an ID
089     */
090    @Override
091    public String getID() {
092        return ID;
093    }
094}
