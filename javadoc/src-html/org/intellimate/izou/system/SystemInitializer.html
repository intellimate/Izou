001package org.intellimate.izou.system;
002
003import org.apache.logging.log4j.LogManager;
004import org.apache.logging.log4j.Logger;
005
006import java.net.InetAddress;
007import java.net.UnknownHostException;
008
009/**
010 * The SystemInitializer does any system wide initialization before Izou is started. An example of this would be setting
011 * system properties relevant to izou etc.
012 */
013public class SystemInitializer {
014    private final Logger fileLogger = LogManager.getLogger(this.getClass());
015
016    /**
017     * Creates a new SystemInitializer object
018     */
019    public SystemInitializer() {
020
021    }
022
023    /**
024     * Initializes the system by calling all init methods
025     */
026    public void initSystem() {
027        setSystemProperties();
028    }
029
030    /**
031     * Sets all system properties relevant for Izou
032     */
033    private void setSystemProperties() {
034        setLocalHostProperty();
035    }
036
037    /**
038     * Gets the local host if it can, and sets it as a system property
039     */
040    private void setLocalHostProperty() {
041        String hostName = "unkown";
042        try {
043            hostName = InetAddress.getLocalHost().getHostName();
044        } catch (UnknownHostException e) {
045            fileLogger.error("Unable to resolve hostname, setting hostname as 'unkown'");
046        }
047
048        System.setProperty("host.name", hostName);
049    }
050}
