001package org.intellimate.izou.system.sound;
002
003import org.intellimate.izou.addon.AddOnModel;
004import org.intellimate.izou.main.Main;
005
006import javax.sound.sampled.*;
007import java.io.IOException;
008
009/**
010 * delegates the Clip-Methods
011 * @author LeanderK
012 * @version 1.0
013 */
014public class IzouSoundLineClip extends IzouSoundDataLine implements Clip {
015    private final Clip clip;
016
017    public IzouSoundLineClip(Clip clip, Main main, boolean isPermanent, AddOnModel addOnModel) {
018        super(clip, main, isPermanent, addOnModel);
019        this.clip = clip;
020    }
021
022    /**
023     * Opens the clip, meaning that it should acquire any required
024     * system resources and become operational.  The clip is opened
025     * with the format and audio data indicated.
026     * If this operation succeeds, the line is marked as open and an
027     * <code>{@link LineEvent.Type#OPEN OPEN}</code> event is dispatched
028     * to the line's listeners.
029     * <p>
030     * Invoking this method on a line which is already open is illegal
031     * and may result in an IllegalStateException.
032     * <p>
033     * Note that some lines, once closed, cannot be reopened.  Attempts
034     * to reopen such a line will always result in a
035     * <code>{@link LineUnavailableException}</code>.
036     *
037     * @param format the format of the supplied audio data
038     * @param data a byte array containing audio data to load into the clip
039     * @param offset the point at which to start copying, expressed in
040     * <em>bytes</em> from the beginning of the array
041     * @param bufferSize the number of <em>bytes</em>
042     * of data to load into the clip from the array.
043     * @throws LineUnavailableException if the line cannot be
044     * opened due to resource restrictions
045     * @throws IllegalArgumentException if the buffer size does not represent
046     * an integral number of sample frames,
047     * or if <code>format</code> is not fully specified or invalid
048     * @throws IllegalStateException if the line is already open
049     * @throws SecurityException if the line cannot be
050     * opened due to security restrictions
051     *
052     * @see #close
053     * @see #isOpen
054     * @see LineListener
055     */
056    @Override
057    public void open(AudioFormat format, byte[] data, int offset, int bufferSize) throws LineUnavailableException {
058        clip.open(format, data, offset, bufferSize);
059    }
060
061    /**
062     * Opens the clip with the format and audio data present in the provided audio
063     * input stream.  Opening a clip means that it should acquire any required
064     * system resources and become operational.  If this operation
065     * input stream.  If this operation
066     * succeeds, the line is marked open and an
067     * <code>{@link LineEvent.Type#OPEN OPEN}</code> event is dispatched
068     * to the line's listeners.
069     * <p>
070     * Invoking this method on a line which is already open is illegal
071     * and may result in an IllegalStateException.
072     * <p>
073     * Note that some lines, once closed, cannot be reopened.  Attempts
074     * to reopen such a line will always result in a
075     * <code>{@link LineUnavailableException}</code>.
076     *
077     * @param stream an audio input stream from which audio data will be read into
078     * the clip
079     * @throws LineUnavailableException if the line cannot be
080     * opened due to resource restrictions
081     * @throws IOException if an I/O exception occurs during reading of
082     * the stream
083     * @throws IllegalArgumentException if the stream's audio format
084     * is not fully specified or invalid
085     * @throws IllegalStateException if the line is already open
086     * @throws SecurityException if the line cannot be
087     * opened due to security restrictions
088     *
089     * @see #close
090     * @see #isOpen
091     * @see LineListener
092     */
093    @Override
094    public void open(AudioInputStream stream) throws LineUnavailableException, IOException {
095        clip.open(stream);
096    }
097
098    /**
099     * Obtains the media length in sample frames.
100     * @return the media length, expressed in sample frames,
101     * or <code>AudioSystem.NOT_SPECIFIED</code> if the line is not open.
102     * @see AudioSystem#NOT_SPECIFIED
103     */
104    @Override
105    public int getFrameLength() {
106        return clip.getFrameLength();
107    }
108
109    /**
110     * Obtains the media duration in microseconds
111     * @return the media duration, expressed in microseconds,
112     * or <code>AudioSystem.NOT_SPECIFIED</code> if the line is not open.
113     * @see AudioSystem#NOT_SPECIFIED
114     */
115    @Override
116    public long getMicrosecondLength() {
117        return clip.getMicrosecondLength();
118    }
119
120    /**
121     * Sets the media position in sample frames.  The position is zero-based;
122     * the first frame is frame number zero.  When the clip begins playing the
123     * next time, it will start by playing the frame at this position.
124     * <p>
125     * To obtain the current position in sample frames, use the
126     * <code>{@link DataLine#getFramePosition getFramePosition}</code>
127     * method of <code>DataLine</code>.
128     *
129     * @param frames the desired new media position, expressed in sample frames
130     */
131    @Override
132    public void setFramePosition(int frames) {
133        clip.setFramePosition(frames);
134    }
135
136    /**
137     * Sets the media position in microseconds.  When the clip begins playing the
138     * next time, it will start at this position.
139     * The level of precision is not guaranteed.  For example, an implementation
140     * might calculate the microsecond position from the current frame position
141     * and the audio sample frame rate.  The precision in microseconds would
142     * then be limited to the number of microseconds per sample frame.
143     * <p>
144     * To obtain the current position in microseconds, use the
145     * <code>{@link DataLine#getMicrosecondPosition getMicrosecondPosition}</code>
146     * method of <code>DataLine</code>.
147     *
148     * @param microseconds the desired new media position, expressed in microseconds
149     */
150    @Override
151    public void setMicrosecondPosition(long microseconds) {
152        clip.setMicrosecondPosition(microseconds);
153    }
154
155    /**
156     * Sets the first and last sample frames that will be played in
157     * the loop.  The ending point must be greater than
158     * or equal to the starting point, and both must fall within the
159     * the size of the loaded media.  A value of 0 for the starting
160     * point means the beginning of the loaded media.  Similarly, a value of -1
161     * for the ending point indicates the last frame of the media.
162     * @param start the loop's starting position, in sample frames (zero-based)
163     * @param end the loop's ending position, in sample frames (zero-based), or
164     * -1 to indicate the final frame
165     * @throws IllegalArgumentException if the requested
166     * loop points cannot be set, usually because one or both falls outside
167     * the media's duration or because the ending point is
168     * before the starting point
169     */
170    @Override
171    public void setLoopPoints(int start, int end) {
172        clip.setLoopPoints(start, end);
173    }
174
175    /**
176     * Starts looping playback from the current position.   Playback will
177     * continue to the loop's end point, then loop back to the loop start point
178     * <code>count</code> times, and finally continue playback to the end of
179     * the clip.
180     * <p>
181     * If the current position when this method is invoked is greater than the
182     * loop end point, playback simply continues to the
183     * end of the clip without looping.
184     * <p>
185     * A <code>count</code> value of 0 indicates that any current looping should
186     * cease and playback should continue to the end of the clip.  The behavior
187     * is undefined when this method is invoked with any other value during a
188     * loop operation.
189     * <p>
190     * If playback is stopped during looping, the current loop status is
191     * cleared; the behavior of subsequent loop and start requests is not
192     * affected by an interrupted loop operation.
193     *
194     * @param count the number of times playback should loop back from the
195     * loop's end position to the loop's  start position, or
196     * <code>{@link #LOOP_CONTINUOUSLY}</code> to indicate that looping should
197     * continue until interrupted
198     */
199    @Override
200    public void loop(int count) {
201        clip.loop(count);
202    }
203}
