012/**
013 * the base class for every IzouSoundLine, provides basic implementation of the Methods defined int IzouSoundLine and
014 * delegates to Line, AutoCloseable etc.
015 * @author LeanderK
016 * @version 1.0
017 */
018public class IzouSoundLineBaseClass extends IzouModule implements Line, AutoCloseable, IzouSoundLine {
019    protected final Line line;
020    private Future<?> closingThread;
021    private boolean isPermanent;
022    protected final SoundManager soundManager;
023    private final AddOnModel addOnModel;
024    private boolean isMuted = false;
025    private Consumer<Void> closeCallback = null;
026    private boolean muteIfNonPermanent = true;
027    private Consumer<Void> muteCallback = null;
028    private Identification responsibleID;
029
030    public IzouSoundLineBaseClass(Line line, Main main, boolean isPermanent, AddOnModel addOnModel) {
031        super(main);
032        this.line = line;
033        this.isPermanent = isPermanent;
034        this.addOnModel = addOnModel;
035        if (!isPermanent) {
036            closingThread = getClosingThread(line, main, addOnModel);
037        } else {
038            closingThread = null;
039        }
040        soundManager = null;
041    }
042
043    private Future<?> getClosingThread(Line line, Main main, AddOnModel addOnModel) {
044        return main.getThreadPoolManager().getIzouThreadPool().submit(() -> {
045            try {
046                Thread.sleep(600000);
047            } catch (InterruptedException e) {
048                error("interrupted while sleeping, canceling");
049                return;
050            }
051            if (line.isOpen()) {
052                debug("closing line " + line + "for Addon " + addOnModel);
053                line.close();
054            }
055        });
056    }

068    /**
069     * Opens the line, indicating that it should acquire any required
070     * system resources and become operational.
071     * If this operation
072     * succeeds, the line is marked as open, and an <code>OPEN</code> event is dispatched
073     * to the line's listeners.
074     * <p>
075     * Note that some lines, once closed, cannot be reopened.  Attempts
076     * to reopen such a line will always result in an <code>LineUnavailableException</code>.
077     * <p>
078     * Some types of lines have configurable properties that may affect
079     * resource allocation.   For example, a <code>DataLine</code> must
080     * be opened with a particular format and buffer size.  Such lines
081     * should provide a mechanism for configuring these properties, such
082     * as an additional <code>open</code> method or methods which allow
083     * an application to specify the desired settings.
084     * <p>
085     * This method takes no arguments, and opens the line with the current
086     * settings.  For <code>SourceDataLine</code> and
087     * <code>TargetDataLine</code> objects, this means that the line is
088     * opened with default settings.  For a <code>Clip</code>, however,
089     * the buffer size is determined when data is loaded.  Since this method does not
090     * allow the application to specify any data to load, an IllegalArgumentException
091     * is thrown. Therefore, you should instead use one of the <code>open</code> methods
092     * provided in the <code>Clip</code> interface to load data into the <code>Clip</code>.
093     * <p>
094     * For <code>DataLine</code>'s, if the <code>DataLine.Info</code>
095     * object which was used to retrieve the line, specifies at least
096     * one fully qualified audio format, the last one will be used
097     * as the default format.
098     *
099     * @throws IllegalArgumentException if this method is called on a Clip instance.
100     * @throws LineUnavailableException if the line cannot be
101     * opened due to resource restrictions.
102     * @throws SecurityException if the line cannot be
103     * opened due to security restrictions.
104     *
105     * @see #close
106     * @see #isOpen
107     */
108    @Override
109    public void open() throws LineUnavailableException {
110        if (!line.isOpen() && !isPermanent && muteCallback != null)
111            muteCallback.accept(null);
112        line.open();
113    }
114
115    /**
116     * Closes the line, indicating that any system resources
117     * in use by the line can be released.  If this operation
118     * succeeds, the line is marked closed and a <code>CLOSE</code> event is dispatched
119     * to the line's listeners.
120     * @throws SecurityException if the line cannot be
121     * closed due to security restrictions.
122     *
123     * @see #open
124     * @see #isOpen
125     */
126    @Override
127    public void close() {
128        if (line.isOpen() && closeCallback != null)
129            closeCallback.accept(null);
130        line.close();
131    }
132
133    /**
134     * Indicates whether the line is open, meaning that it has reserved
135     * system resources and is operational, although it might not currently be
136     * playing or capturing sound.
137     * @return <code>true</code> if the line is open, otherwise <code>false</code>
138     *
139     * @see #open()
140     * @see #close()
141     */
142    @Override
143    public boolean isOpen() {
144        return line.isOpen();
145    }

194    /**
195     * returns whether the line is permanently-available.
196     * If a line is not permanently available, it will close after max. 10 minutes
197     * @return true if permanent.
198     */
199    @Override
200    public boolean isPermanent() {
201        return isPermanent;
202    }
203
204    void setToPermanent() {
205        if (isPermanent)
206            return;
207        closingThread.cancel(true);
208        closingThread = null;
209        isPermanent = true;
210    }
211
212    void setToNonPermanent() {
213        if (!isPermanent)
214            return;
215        closingThread = getClosingThread(line, main, addOnModel);
216        isPermanent = false;
217    }

253    /**

264    /**
265     * retruns whether other Addons audio-inputs should be muted while this line is open (only works for non-permanent lines).
266     * @return true if muted, false if not
267     */
268    @Override
269    public boolean isMuteIfNonPermanent() {
270        return muteIfNonPermanent;
271    }
272
273    void setMuted(boolean isMuted) {
274        BooleanControl bc = (BooleanControl) line.getControl(BooleanControl.Type.MUTE);
275        if (bc != null) {
276            bc.setValue(isMuted); // true to mute the line, false to unmute
277        }
278        this.isMuted = isMuted;
279    }
280
281    void registerCloseCallback(Consumer<Void> consumer) {
282        this.closeCallback = consumer;
283    }
284
285    void registerMuteCallback(Consumer<Void> consumer) {
286        this.muteCallback = consumer;
287    }
