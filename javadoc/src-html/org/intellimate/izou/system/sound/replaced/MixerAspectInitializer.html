001package org.intellimate.izou.system.sound.replaced;
002
003import org.apache.logging.log4j.LogManager;
004import org.apache.logging.log4j.Logger;
005import org.intellimate.izou.addon.AddOnModel;
006import org.intellimate.izou.main.Main;
007import org.intellimate.izou.security.exceptions.IzouPermissionException;
008import org.intellimate.izou.system.sound.*;
009
010import javax.sound.sampled.Clip;
011import javax.sound.sampled.DataLine;
012import javax.sound.sampled.Line;
013import javax.sound.sampled.SourceDataLine;
014
015/**
016 * this class is only used to connect the MixerAspect to Izou (not able to instantiate it).
017 * @author LeanderK
018 * @version 1.0
019 */
020public class MixerAspectInitializer {
021    static Main main = null;
022    private static final Logger logger= LogManager.getLogger(MixerAspectInitializer.class);
023
024    public static synchronized void init(Main main) {
025        if (MixerAspectInitializer.main == null)
026            MixerAspectInitializer.main = main;
027    }
028
029    /**
030     * creates the appropriate IzouSoundLine if the request originates from an AddOn.
031     * @param line the line
032     * @return an IzouSoundLine if an addon requested the line
033     */
034    static Line getAndRegisterLine(Line line) {
035        AddOnModel addOnModel;
036        try {
037            addOnModel = main.getSecurityManager().getOrThrowAddOnModelForClassLoader();
038        } catch (IzouPermissionException e) {
039            logger.debug("the SoundManager will not manage this line, obtained by system");
040            return line;
041        }
042        IzouSoundLineBaseClass izouSoundLine;
043        if (line instanceof SourceDataLine) {
044            if (line instanceof Clip) {
045                izouSoundLine = new IzouSoundLineClipAndSDLine((Clip) line, (SourceDataLine) line, main, false, addOnModel);
046            } else {
047                izouSoundLine = new IzouSoundSourceDataLine((SourceDataLine) line, main, false, addOnModel);
048            }
049        } else if (line instanceof Clip) {
050            izouSoundLine = new IzouSoundLineClip((Clip) line, main, false, addOnModel);
051        } else if (line instanceof DataLine) {
052            izouSoundLine = new IzouSoundDataLine((DataLine) line, main, false, addOnModel);
053        } else {
054            izouSoundLine = new IzouSoundLineBaseClass(line, main, false, addOnModel);
055        }
056        main.getSoundManager().addIzouSoundLine(addOnModel, izouSoundLine);
057        return izouSoundLine;
058    }
059}
