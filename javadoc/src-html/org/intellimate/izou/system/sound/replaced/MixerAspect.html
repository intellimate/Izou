001package org.intellimate.izou.system.sound.replaced;
002
003import org.apache.logging.log4j.LogManager;
004import org.apache.logging.log4j.Logger;
005import org.aspectj.lang.ProceedingJoinPoint;
006import org.aspectj.lang.annotation.Around;
007import org.aspectj.lang.annotation.Aspect;
008import org.intellimate.izou.addon.AddOnModel;
009import org.intellimate.izou.main.Main;
010import org.intellimate.izou.security.exceptions.IzouPermissionException;
011import org.intellimate.izou.system.sound.*;
012
013import javax.sound.sampled.Clip;
014import javax.sound.sampled.DataLine;
015import javax.sound.sampled.Line;
016import javax.sound.sampled.SourceDataLine;
017
018/**
019 * @author LeanderK
020 * @version 1.0
021 */
022@Aspect
023public class MixerAspect {
024    static Main main;
025    private static final Logger logger= LogManager.getLogger(MixerAspectInitializer.class);
026
027    public static synchronized void init(Main main) {
028        if (MixerAspectInitializer.main == null)
029            MixerAspectInitializer.main = main;
030    }
031
032    /**
033     * creates the appropriate IzouSoundLine if the request originates from an AddOn.
034     * @param line the line
035     * @return an IzouSoundLine if an addon requested the line
036     */
037    static Line getAndRegisterLine(Line line) {
038        AddOnModel addOnModel;
039        try {
040            addOnModel = main.getSecurityManager().getOrThrowAddOnModelForClassLoader();
041        } catch (IzouPermissionException e) {
042            logger.debug("the SoundManager will not manage this line, obtained by system");
043            return line;
044        }
045        IzouSoundLineBaseClass izouSoundLine;
046        if (line instanceof SourceDataLine) {
047            if (line instanceof Clip) {
048                izouSoundLine = new IzouSoundLineClipAndSDLine((Clip) line, (SourceDataLine) line, main, false, addOnModel);
049            } else {
050                izouSoundLine = new IzouSoundSourceDataLine((SourceDataLine) line, main, false, addOnModel);
051            }
052        } else if (line instanceof Clip) {
053            izouSoundLine = new IzouSoundLineClip((Clip) line, main, false, addOnModel);
054        } else if (line instanceof DataLine) {
055            izouSoundLine = new IzouSoundDataLine((DataLine) line, main, false, addOnModel);
056        } else {
057            izouSoundLine = new IzouSoundLineBaseClass(line, main, false, addOnModel);
058        }
059        main.getSoundManager().addIzouSoundLine(addOnModel, izouSoundLine);
060        return izouSoundLine;
061    }
062
063    @Around("execution(* javax.sound.sampled.AudioSystem.getLine(javax.sound.sampled.Line.Info))")
064    public Object getLineAdvice(ProceedingJoinPoint pjp) throws Throwable {
065        System.out.println("request to add line");
066        pjp.getArgs();
067        Line line = (Line) pjp.proceed();
068        return MixerAspectInitializer.getAndRegisterLine(line);
069    }
070}
