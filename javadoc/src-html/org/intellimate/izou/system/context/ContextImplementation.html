046    /**
047     * creates a new context for the addOn
048     *
049     * A context contains all the "global" or generally necessary information an addOn might need that it otherwise does
050     * not have access too
051     *
052     * @param addOn the addOn for which to create a new context
053     * @param main instance of main
054     * @param logLevel the logLevel to initialize the IzouLogger with
055     */
056    public ContextImplementation(AddOnModel addOn, Main main, String logLevel) {
057        this.addOn = addOn;
058        this.main = main;
059        this.files = new FilesImpl();
060        this.threadPool = new ThreadPoolImpl();
061
062        IzouLogger izouLogger = main.getIzouLogger();
063        ExtendedLogger logger = null;
064        if (izouLogger != null)
065            this.logger = izouLogger.createFileLogger(addOn.getID(), logLevel);
066        else {
067            this.logger = null;
068            org.apache.logging.log4j.Logger fileLogger = LogManager.getLogger(this.getClass());
069            fileLogger.error("IzouLogger has not been initialized");
070            throw new NullPointerException("IzouLogger has not been initialized");
071        }
072    }

149        /**
150         * Use this method to register a file with the watcherService
151         *
152         * @param dir directory of file
153         * @param fileType the name/extension of the file
154         *                 IMPORTANT: Please try to always enter the full name with extension of the file (Ex: "test.txt"),
155         *                 it would be best if the fileType is the full file name, and that the file name is clearly
156         *                 distinguishable from other files.
157         *                 For example, the property files are stored with the ID of the addon they belong too. That way
158         *                 every property file is easily distinguishable.
159         * @param reloadableFile object of interface that file belongs to
160         * @throws IOException exception thrown by watcher service
161         */
162        @Override
163        public void registerFileDir(Path dir, String fileType, ReloadableFile reloadableFile) throws IOException {
164            main.getFileManager().registerFileDir(dir, fileType, reloadableFile);
165        }

167        /**
168         * Writes default file to real file
169         * The default file would be a file that can be packaged along with the code, from which a real file (say a
170         * properties file for example) can be loaded. This is useful because there are files (like property files0 that
171         * cannot be shipped with the package and have to be created at runtime. To still be able to fill these files, you
172         * can create a default file (usually txt) from which the content, as mentioned above, can then be loaded into the
173         * real file.
174         *
175         * @param defaultFilePath path to default file (or where it should be created)
176         * @param realFilePath path to real file (that should be filled with content of default file)
177         * @return true if operation has succeeded, else false
178         */
179        @Override

181            return main.getFileManager().writeToFile(defaultFilePath, realFilePath);

474             * @return An Optional Object which may or may not contains an EventPublisher

482            /**
483             * with this method you can unregister EventPublisher add a Source of Events to the System.
484             * <p>
485             * This method represents a higher level of abstraction! Use the EventManager to fire Events!
486             * This method is intended for use cases where you have an entire new source of events (e.g. network)
487             * @param identification the Identification of the Source
488             */
489            @Override
490            public void unregisterEventPublisher(Identification identification) {
491                main.getEventDistributor().unregisterEventPublisher(identification);
492            }

494            /**
495             * Registers an EventController to control EventDispatching-Behaviour
496             * <p>
497             * Method is thread-safe.
498             * It is expected that this method executes quickly.
499             *
500             * @param eventsController the EventController Interface to control event-dispatching
501             * @throws IllegalIDException not yet implemented
502             */
503            @Override
504            public void registerEventsController(EventsControllerModel eventsController) throws IllegalIDException {
505                main.getEventDistributor().registerEventsController(eventsController);
506            }

508            /**
509             * Unregisters an EventController
510             * <p>
511             * Method is thread-safe.
512             *
513             * @param eventsController the EventController Interface to remove
514             */
515            @Override
516            public void unregisterEventsController(EventsControllerModel eventsController) {
517                main.getEventDistributor().unregisterEventsController(eventsController);
518            }
519
520            /**
521             * fires the event concurrently, this is generally discouraged.
522             * <p>
523             * This method should not be used for normal Events, for for events which obey the following laws:<br>
524             * 1. they are time critical.<br>
525             * 2. addons are not expected to react in any way beside a small update<br>

527             * if your event matches the above laws, you may consider firing it concurrently.
