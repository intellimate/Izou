029/**
030 * Manages all the AddOns.
031 */
032public class AddOnManager extends IzouModule implements AddonThreadPoolUser {
033    private IdentifiableSet<AddOnModel> addOns = new IdentifiableSet<>();
034    private HashMap<AddOnModel, PluginWrapper> pluginWrappers = new HashMap<>();
035    private Set<URL> aspects = new HashSet<>();
036    private Set<URL> affected = new HashSet<>();

038    public AddOnManager(Main main) {
039        super(main);

042    /**
043    * retrieves and registers all AddOns.
044    */
045    public void retrieveAndRegisterAddOns() {
046        addOns.addAll(loadAddOns());
047        registerAllAddOns(addOns);
048    }

050    /**
051     * Adds AddOns without registering them.
052     * @param addOns a List containing all the AddOns
053     */
054    public void addAddOnsWithoutRegistering(List<AddOnModel> addOns) {
055        this.addOns.addAll(addOns);
056    }
057
058    /**
059     * registers all AddOns.
060     *
061     * @param addOns a List containing all the AddOns
062     */
063    public void addAndRegisterAddOns(List<AddOnModel> addOns) {
064        this.addOns.addAll(addOns);
065        registerAllAddOns(this.addOns);
066    }
067    
068    public void registerAllAddOns(IdentifiableSet<AddOnModel> addOns) {
069        initAddOns(addOns);
070        List<CompletableFuture<Void>> futures = addOns.stream()
071                .map(addOn -> submit((Runnable) addOn::register))
072                .collect(Collectors.toList());
073        try {
074            timeOut(futures, 3000);
075        } catch (InterruptedException e) {
076            debug("interrupted while trying to time out the addOns", e);
077        }
078    }
079
080    private void initAddOns(IdentifiableSet<AddOnModel> addOns) {
081        List<CompletableFuture<Void>> futures = addOns.stream()
082                .map(addOn -> {
083                    Context context = new ContextImplementation(addOn, main, Level.DEBUG.name());
084                    return submit(() -> addOn.initAddOn(context));
085                })
086                .collect(Collectors.toList());
087        try {
088            timeOut(futures, 1000);
089        } catch (InterruptedException e) {
090            debug("interrupted while trying to time out the addOns", e);
091        }
092    }
093
094    /**
095     * This method searches all the "/lib"-directory for AddOns and adds them to the addOnList
096     * @return the retrieved addOns
097     */
098    private List<AddOnModel> loadAddOns() {
099        debug("searching for addons in: " + getMain().getFileSystemManager().getLibLocation());
100        PluginManager pluginManager = new DefaultPluginManager(getMain().getFileSystemManager().getLibLocation(), new ArrayList<>(aspects), new ArrayList<>(affected));
101        // load the plugins
102        debug("loading plugins");
103        pluginManager.loadPlugins();
104        debug("loaded: " + pluginManager.getPlugins().toString());
105
106        // start (active/resolved) the plugins
107        try {
108            debug("starting plugins");
109            pluginManager.startPlugins();
110        } catch (Exception | NoClassDefFoundError e) {
111            error("Error while trying to start the PF4J-Plugins", e);
112        }
113        try {
114            debug("retrieving addons from the plugins");
115            List<AddOnModel> addOns = pluginManager.getExtensions(AddOnModel.class);
116            debug("retrieved: " + addOns.toString());
117            KeyManager keyManager = new KeyManager();
118            addOns.stream()
119                    .filter(addOn -> addOn.getClass().getClassLoader() instanceof IzouPluginClassLoader)
120                    .forEach(addOn -> {
121                        IzouPluginClassLoader izouPluginClassLoader = (IzouPluginClassLoader) addOn.getClass()
122                                .getClassLoader();
123                        PluginWrapper plugin = pluginManager.getPlugin(izouPluginClassLoader.getPluginDescriptor()
124                                .getPluginId());
125                        keyManager.manageAddOnKey(plugin.getDescriptor());
126                        pluginWrappers.put(addOn, plugin);
127                    });
128            keyManager.saveAddOnKeys();
129            return addOns;
130        } catch (Exception e) {
131            log.fatal("Error while trying to start the AddOns", e);
132            return new ArrayList<>();
133        }
134    }

136    /**
137     * returns the addOn loaded from the ClassLoader
138     * @param classLoader the classLoader
139     * @return the (optional) AddOnModel
140     */
141    public Optional<AddOnModel> getAddOnForClassLoader(ClassLoader classLoader) {
142        return addOns.stream()
143                .filter(addOnModel -> addOnModel.getClass().getClassLoader().equals(classLoader))
144                .findFirst();
145    }

147    /**
148     * returns the (optional) PluginWrapper for the AddonModel.
149     * If the return is empty, it means that the AddOn was not loaded through pf4j
150     * @param addOnModel the AddOnModel
151     * @return the PluginWrapper if loaded through pf4j or empty if added as an argument
152     */
153    public Optional<PluginWrapper> getPluginWrapper(AddOnModel addOnModel) {
154        return Optional.of(pluginWrappers.get(addOnModel));
155    }

157    /**
158     * checks whether the AddOn was loaded through pf4j
159     * @param addOnModel the AddOnModel to check
160     * @return true if loaded, false if not
161     */
162    public boolean loadedThroughPF4J (AddOnModel addOnModel) {
163        return pluginWrappers.get(addOnModel) != null;
164    }

182    /**
183     * The KeyManager in the AddOnManager loads or creates a {@link SecretKey} for each AddOn at addOn load time,
184     * depending if one already exists. It then distributes them to each addOn being loaded, and then finaly saves them
185     * again.
186     * <p>
187     *     This is necessary for the {@link org.intellimate.izou.security.storage.SecureStorage} in order to save
188     *     data matching to each addOn. This secret key serves as key to the data of an addOn being saved. In other
189     *     words, each addOn data is matched with the secret key instead of the plugin descriptor itself in order to
190     *     avoid serialization of the addon descriptor, which would entail a huge mess. So the secret key of each addOn
191     *     is pretty much a "signature" of each addOn, easily identifying it.
192     * </p>
193     */
194    private class KeyManager {
195        private HashMap<String, SecretKey> addOnKeys;
196        boolean changed;
197
198        /**
199         * Creates a new KeyManager object
200         */
201        private KeyManager() {
202            addOnKeys = new HashMap<>();
203            retrieveAddonKeys();
204        }
205
206        /**
207         * Check if a SecretKey already exists for the plugin descriptor, if not creates a new one, and then gives it to
208         * the plugin descriptor.
209         * @param descriptor The plugin descriptor to give a SecretKey
210         */
211        private void manageAddOnKey(PluginDescriptor descriptor) {
212            SecretKey secretKey = addOnKeys.get(descriptor.getPluginId());
213
214            if (secretKey == null) {
215                SecurityModule module = new SecurityModule();
216                secretKey = module.generateKey();
217                addOnKeys.put(descriptor.getPluginId(), secretKey);
218                changed = true;
219            }
220
221            descriptor.setSecureID(secretKey);
222        }
223
224        /**
225         * Retrieves all saved addOnKeys (they cannot change, since there are dependencies on them, so they are saved
226         * and retrieved)
227         */
228        private void retrieveAddonKeys() {
229            changed = false;
230
231            try {
232                String workingDir = FileSystemManager.FULL_WORKING_DIRECTORY;
233                final String keyStoreFile = workingDir + File.separator + "system" + File.separator + "data"
234                        + File.separator + "addon_keys.keystore";
235                KeyStore keyStore = createKeyStore(keyStoreFile, "4b[X:+H4CS&avY<)");
236
237                KeyStore.PasswordProtection keyPassword = new KeyStore.PasswordProtection("Ev45j>eP}QTR?K9_"
238                        .toCharArray());
239                Enumeration<String> aliases = keyStore.aliases();
240                while (aliases.hasMoreElements()) {
241                    String alias = aliases.nextElement();
242                    KeyStore.Entry entry = keyStore.getEntry(alias, keyPassword);
243                    SecretKey key = ((KeyStore.SecretKeyEntry) entry).getSecretKey();
244                    addOnKeys.put(alias, key);
245                }
246            } catch(NullPointerException e) {
247                return;
248            } catch (UnrecoverableEntryException | NoSuchAlgorithmException | KeyStoreException e) {
249                error("Unable to retrieve key", e);
250            }
251        }
252
253        /**
254         * Save all addOnKeys in the instance variable {@code addOnKeys} in a keystore
255         */
256        private void saveAddOnKeys() {
257            if (!changed) {
258                return;
259            }
260
261            String workingDir = FileSystemManager.FULL_WORKING_DIRECTORY;
262            final String keyStoreFile = workingDir + File.separator + "system" + File.separator + "data"
263                    + File.separator + "addon_keys.keystore";
264            KeyStore keyStore = createKeyStore(keyStoreFile, "4b[X:+H4CS&avY<)");
265
266            for (String mapKey : addOnKeys.keySet()) {
267                try {
268                    KeyStore.SecretKeyEntry keyStoreEntry = new KeyStore.SecretKeyEntry(addOnKeys.get(mapKey));
269                    KeyStore.PasswordProtection keyPassword = new KeyStore.PasswordProtection("Ev45j>eP}QTR?K9_"
270                            .toCharArray());
271                    keyStore.setEntry(mapKey, keyStoreEntry, keyPassword);
272                } catch (KeyStoreException e) {
273                    error("Unable to store key", e);
274                }
275            }
276
277            try {
278                keyStore.store(new FileOutputStream(keyStoreFile), "4b[X:+H4CS&avY<)".toCharArray());
279            } catch (KeyStoreException | IOException | CertificateException | NoSuchAlgorithmException e) {
280                error("Unable to store key", e);
281            }
282        }
283
284        /**
285         * Creates a new keystore for addOn secret keys
286         *
287         * @param fileName the path to the keystore
288         * @param password the password to use with the keystore
289         * @return the newly created keystore
290         */
291        private KeyStore createKeyStore(String fileName, String password)  {
292            File file = new File(fileName);
293            KeyStore keyStore = null;
294            try {
295                keyStore = KeyStore.getInstance("JCEKS");
296                if (file.exists()) {
297                    keyStore.load(new FileInputStream(file), password.toCharArray());
298                } else {
299                    keyStore.load(null, null);
300                    keyStore.store(new FileOutputStream(fileName), password.toCharArray());
301                }
302            } catch (CertificateException | IOException | KeyStoreException | NoSuchAlgorithmException e) {
303                error("Unable to create key store", e);
304            }
305
306            return keyStore;
307        }
308    }
