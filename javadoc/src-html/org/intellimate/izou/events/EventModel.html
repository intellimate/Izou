001package org.intellimate.izou.events;
002
003import org.intellimate.izou.identification.Identifiable;
004import org.intellimate.izou.identification.Identification;
005import org.intellimate.izou.resource.ListResourceProvider;
006import org.intellimate.izou.resource.ResourceModel;
007
008import java.util.List;
009
010/**
011 * This class represents an Event, the main communication form for the AddOns.
012 * After the call to finalizeEvent() the properties (except adding resources) should be immutable.
013 * @author Leander Kurscheidt
014 * @version 1.0
015 */
016public interface EventModel<X extends EventModel> extends Identifiable {
017    /**
018     * The type of the Event.
019     * It describes the Type of the Event.
020     * @return A String containing an ID
021     */
022    String getType();
023
024    /**
025     * Returns the source of the event, e.g. the object who fired it.
026     *
027     * @return an identifiable
028     */
029    Identification getSource();
030
031    /**
032     * Returns all the resources the event currently has
033     *
034     * @return an instance of ListResourceContainer
035     */
036    ListResourceProvider getListResourceContainer();
037
038    /**
039     * adds a resource to the container
040     * @param resource an instance of the resource to add
041     * @return the resulting Event (which is the same instance)
042     */
043    X addResource(ResourceModel resource);
044
045    /**
046     * adds a list of resources to the container
047     * @param resources a list containing all the resources
048     * @return the resulting Event (which is the same instance)
049     */
050    X addResources(List<ResourceModel> resources);
051
052    /**
053     * returns a list containing all the descriptors.
054     * @return a list containing the descriptors
055     */
056    List<String> getDescriptors();
057
058    /**
059     * returns a List containing all the Descriptors and the type.
060     * @return a List containing the Descriptors
061     */
062    List<String> getAllInformations();
063
064    /**
065     * returns whether the event contains the specific descriptor.
066     * this method also checks whether it matches the type.
067     * @param descriptor a string with the ID of the descriptor
068     * @return boolean when the event contains the descriptor, false when not.
069     */
070    boolean containsDescriptor(String descriptor);
071
072    /**
073     * returns the associated EventBehaviourController
074     * @return an instance of EventBehaviourController
075     */
076    EventBehaviourControllerModel getEventBehaviourController();
077}
