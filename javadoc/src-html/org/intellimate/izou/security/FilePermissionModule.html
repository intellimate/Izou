001package org.intellimate.izou.security;
002
003import com.google.common.io.Files;
004import org.intellimate.izou.addon.AddOnModel;
005import org.intellimate.izou.main.Main;
006import org.intellimate.izou.security.exceptions.IzouPermissionException;
007import org.intellimate.izou.system.file.FileSystemManager;
008
009import java.io.File;
010import java.io.FilePermission;
011import java.io.IOException;
012import java.security.Permission;
013import java.util.ArrayList;
014import java.util.Arrays;
015import java.util.List;
016import java.util.regex.Matcher;
017import java.util.regex.Pattern;
018
019/**
020 * The FilePermissionModule is used to check the access to files
021 * @author LeanderK
022 * @version 1.0
023 */
024public class FilePermissionModule extends PermissionModule {
025    private final List<String> allowedReadDirectories;
026    private final List<String> allowedReadFiles;
027    private final List<File> allowedWriteDirectories;
028    private final List<File> forbiddenWriteDirectories;
029    private final String allowedReadFileTypesRegex;
030    private final String allowedWriteFileTypesRegex;
031
032    /**
033     * Creates a new PermissionModule
034     *
035     * @param main the instance of main
036     */
037    FilePermissionModule(Main main, SecurityManager securityManager) {
038        super(main, securityManager);
039        allowedReadDirectories = new ArrayList<>();
040        allowedReadFiles = new ArrayList<>();
041        forbiddenWriteDirectories = new ArrayList<>();
042        forbiddenWriteDirectories.add(main.getFileSystemManager().getLibLocation());
043        allowedReadFiles.add("/dev/random");
044        allowedReadFiles.add("/dev/urandom");
045        allowedWriteDirectories = new ArrayList<>();
046        allowedReadFileTypesRegex = "(txt|properties|xml|class|json|zip|ds_store|mf|jar|idx|log|dylib|mp3|dylib|certs|"
047                + "so)";
048        allowedWriteFileTypesRegex = "(txt|properties|xml|json|idx|log)";
049        String workingDir = FileSystemManager.FULL_WORKING_DIRECTORY;
050        allowedReadDirectories.add(workingDir);
051        allowedReadDirectories.addAll(Arrays.asList(System.getProperty("java.ext.dirs").split(":")));
052        allowedReadDirectories.add(System.getProperty("java.home"));
053        allowedReadDirectories.add(System.getProperty("user.home"));
054        allowedWriteDirectories.add(main.getFileSystemManager().getLogsLocation());
055        allowedWriteDirectories.add(main.getFileSystemManager().getPropertiesLocation());
056        allowedWriteDirectories.add(main.getFileSystemManager().getResourceLocation());
057    }
058
059    /**
060     * returns true if able to check permissions
061     *
062     * @param permission the permission to check
063     * @return true if able to, false if not
064     */
065    @Override
066    public boolean canCheckPermission(Permission permission) {
067        return permission instanceof FilePermission;
068    }
069
070    /**
071     * Checks if the given addOn is allowed to access the requested service and registers them if not yet registered.
072     *
073     * @param permission the Permission to check
074     * @param addon      the identifiable to check
075     * @throws IzouPermissionException thrown if the addOn is not allowed to access its requested service
076     */
077    @Override
078    public void checkPermission(Permission permission, AddOnModel addon) throws IzouPermissionException {
079        String canonicalName = permission.getName().intern().toLowerCase();
080        String canonicalAction =  permission.getActions().intern().toLowerCase();
081
082        if (canonicalName.contains("all files") || canonicalAction.equals("execute")) {
083            throw getException(permission.getName());
084        }
085        if (canonicalAction.equals("read")) {
086            fileReadCheck(canonicalName);
087        }
088        fileWriteCheck(canonicalName, addon);
089    }
090
091    /**
092     * Determines if the file at the given file path is safe to read from in all aspects, if so returns true, else false
093     *
094     * @param filePath the path to the file to read from
095     */
096    void fileReadCheck(String filePath) {
097        File potentialFile = new File(filePath);
098        String canonicalPath;
099        try {
100            canonicalPath = potentialFile.getCanonicalPath();
101        } catch (IOException e) {
102            error("Error getting canonical path", e);
103            throw getException(filePath);
104        }
105
106        for (String file : allowedReadFiles) {
107            if (canonicalPath.contains(file)) {
108                return;
109            }
110        }
111
112        boolean allowedDirectory = false;
113        for (String dir : allowedReadDirectories) {
114            if (canonicalPath.contains(dir)) {
115                allowedDirectory = true;
116                break;
117            }
118        }
119        if (!allowedDirectory) {
120            throw getException(filePath);
121        }
122
123        String[] pathParts = canonicalPath.split(File.separator);
124        String lastPathPart = pathParts[pathParts.length - 1].toLowerCase();
125
126        String[] pathPeriodParts = lastPathPart.split("\\.");
127        String fileExtension = pathPeriodParts[pathPeriodParts.length - 1].toLowerCase();
128
129        if (!getSecurityManager().getSecureAccess().checkForExistingFileOrDirectory(canonicalPath)
130                || getSecurityManager().getSecureAccess().checkForDirectory(canonicalPath)) {
131            return;
132        }
133
134        Pattern pattern = Pattern.compile(allowedReadFileTypesRegex);
135        Matcher matcher = pattern.matcher(fileExtension);
136        if (!matcher.matches() || fileExtension.equals(lastPathPart))
137            throw getException(filePath);
138    }
139
140    /**
141     * Determines if the file at the given file path is safe to write to in all aspects, if so returns true, else false
142     *
143     * @param filePath the path to the file to write to
144     * @param addOnModel the AddonModel
145     */
146    void fileWriteCheck(String filePath, AddOnModel addOnModel) {
147        File request;
148        try {
149            request =  new File(filePath).getCanonicalFile();
150        } catch (IOException e) {
151            error("Error getting canonical path", e);
152            throw getException(filePath);
153        }
154
155        isForbidden(request, addOnModel);
156
157        if (allowedWriteDirectories.stream()
158                .noneMatch(compare -> request.toPath().startsWith(compare.toPath()))) {
159            throw getException(filePath);
160        }
161
162        if (!getSecurityManager().getSecureAccess().checkForExistingFileOrDirectory(request.toString())
163                || getSecurityManager().getSecureAccess().checkForDirectory(request.toString())) {
164            return;
165        }
166
167        Pattern pattern = Pattern.compile(allowedWriteFileTypesRegex);
168        Matcher matcher = pattern.matcher(Files.getFileExtension(request.toString()));
169        if (!matcher.matches())
170            throw getException(filePath);
171    }
172
173    /**
174     * throws an Exception if the
175     * @param request the requested File
176     * @param addOnModel the AddonModel
177     */
178    private void isForbidden(File request, AddOnModel addOnModel) {
179        if (forbiddenWriteDirectories.stream()
180                .anyMatch(compare -> request.toPath().startsWith(compare.toPath()))) {
181            throw new IzouPermissionException("file: " + request.toString() + " is forbidden. Attempt made by: "
182                    + addOnModel.getID());
183        }
184    }
185}
