001package org.intellimate.izou.security.storage;
002
003import org.apache.logging.log4j.LogManager;
004import org.apache.logging.log4j.Logger;
005import org.intellimate.izou.security.SecurityModule;
006import org.intellimate.izou.system.file.FileSystemManager;
007import ro.fortsoft.pf4j.PluginDescriptor;
008
009import javax.crypto.SecretKey;
010import java.io.*;
011import java.security.KeyStore;
012import java.security.KeyStoreException;
013import java.security.NoSuchAlgorithmException;
014import java.security.UnrecoverableEntryException;
015import java.security.cert.CertificateException;
016import java.util.HashMap;
017
018/**
019 * The SecureStorage class offers a way for addOns to store data so that other addOns cannot access it. For example if
020 * addOn A wants to store the users username and password to some service, it can do so using this class without any
021 * other addOn having access to that information.
022 * <p>
023 *     While the stored information is encrypted, it is not safe from the user. In theory and with a lot of effort, the
024 *     user could extract the keys and decrypt the stored information since the keys are not hidden from the user (only
025 *     from addOns).
026 * </p>
027 */
028//TODO: @Julian why doesn't it extend IzouModule?
029public final class SecureStorage {
030    private static boolean exists = false;
031    private HashMap<SecretKey, SecureContainer> containers;
032    private final Logger logger = LogManager.getLogger(this.getClass());
033
034    /**
035     * Creates an SecureStorage. There can only be one single SecureStorage, so calling this method twice
036     * will cause an illegal access exception.
037     *
038     * @return a SecureAccess object
039     * @throws IllegalAccessException thrown if this method is called more than once
040     */
041    public static SecureStorage createSecureStorage() throws IllegalAccessException {
042        if (!exists) {
043            SecureStorage secureStorage = new SecureStorage();
044            exists = true;
045            return secureStorage;
046        }
047
048        throw new IllegalAccessException("Cannot create more than one instance of IzouSecurityManager");
049    }
050
051    /**
052     * Creates a new SecureStorage instance if and only if none has been created yet
053     *
054     * @throws IllegalAccessException thrown if this method is called more than once
055     */
056    public SecureStorage() throws IllegalAccessException, NullPointerException {
057        if (exists) {
058            throw new IllegalAccessException("Cannot create more than one instance of IzouSecurityManager");
059        }
060
061        SecretKey key = retrieveKey();
062        if (key == null) {
063            SecurityModule securityModule = new SecurityModule();
064            key = securityModule.generateKey();
065
066            if (key != null) {
067                storeKey(key);
068            } else {
069                throw new NullPointerException("Unable to create security key");
070            }
071        }
072
073        containers = retrieveContainers();
074        if (containers == null) {
075            containers = new HashMap<>();
076        }
077    }
078
079    /**
080     * Stores a {@link SecureContainer} with the given secure ID of the plugin descriptor. Each addOn can only have 1
081     * secure container, so in order to update it, retrieve it and store it again.
082     *
083     * @param descriptor The plugin descriptor belonging to an addOn
084     * @param container The secure container to be stored with an addOn
085     */
086    public void store(PluginDescriptor descriptor, SecureContainer container) {
087        HashMap<String, String> clearTextData = container.getClearTextData();
088        HashMap<byte[], byte[]> cryptData = container.getCryptData();
089
090        SecretKey secretKey = retrieveKey();
091        SecurityModule module = new SecurityModule();
092        for (byte[] key : cryptData.keySet()) {
093            clearTextData.put(module.decryptAES(key, secretKey), module.decryptAES(cryptData.get(key), secretKey));
094            cryptData.remove(key);
095        }
096
097        container.setCryptData(cryptData);
098        containers.put(descriptor.getSecureID(), container);
099        saveContainers();
100    }
101
102    /**
103     * Retrieves a {@link SecureContainer} with the given secure ID of the plugin descriptor
104     *
105     * @param descriptor The plugin descriptor belonging to an addOn
106     * @return  container The secure container that was retrieved
107     */
108    public SecureContainer retrieve(PluginDescriptor descriptor) {
109        SecureContainer container = containers.get(descriptor.getSecureID());
110        HashMap<byte[], byte[]> cryptData = container.getCryptData();
111        HashMap<String, String> clearTextData = container.getClearTextData();
112
113        SecretKey secretKey = retrieveKey();
114        SecurityModule module = new SecurityModule();
115        for (byte[] key : cryptData.keySet()) {
116            clearTextData.put(module.decryptAES(key, secretKey), module.decryptAES(cryptData.get(key), secretKey));
117            cryptData.remove(key);
118        }
119
120        container.setClearTextData(clearTextData);
121        return container;
122    }
123
124    /**
125     * Saves the containers to ./system/data/containers.ser
126     */
127    private void saveContainers() {
128        String workingDir = FileSystemManager.FULL_WORKING_DIRECTORY;
129        final String containerFile = workingDir + File.separator + "system" + File.separator + "data" + File.separator
130                + "containers.ser";
131        try {
132            FileOutputStream fileOut = new FileOutputStream(containerFile);
133            ObjectOutputStream out = new ObjectOutputStream(fileOut);
134            out.writeObject(containers);
135            out.close();
136            fileOut.close();
137        } catch(IOException e) {
138            logger.error("Unable to save containers to file", e);
139        }
140    }
141
142    /**
143     * Retrieves the containers from ./system/data/containers.ser if the file is found, else returns null
144     *
145     * @return the containers from ./system/data/containers.ser if the file is found, else null
146     */
147    private HashMap<SecretKey, SecureContainer> retrieveContainers() {
148        HashMap<SecretKey, SecureContainer> containers = null;
149        String workingDir = FileSystemManager.FULL_WORKING_DIRECTORY;
150        final String containerFile = workingDir + File.separator + "system" + File.separator + "data" + File.separator
151                + "containers.ser";
152        try {
153            FileInputStream fileIn = new FileInputStream(containerFile);
154            ObjectInputStream in = new ObjectInputStream(fileIn);
155            Object o = in.readObject();
156            if (o instanceof HashMap) {
157                containers = (HashMap) o;
158            }
159            in.close();
160            fileIn.close();
161        } catch (FileNotFoundException e) {
162            return null;
163        } catch(IOException | ClassNotFoundException e) {
164            logger.error("Unable to retrieve containers from file", e);
165        }
166
167        return containers;
168    }
169
170    /**
171     * Retrieves the izou aes key stored in a keystore
172     *
173     * @return the izou aes key stored in a keystore
174     */
175    private SecretKey retrieveKey() {
176        SecretKey key = null;
177        try {
178            String workingDir = FileSystemManager.FULL_WORKING_DIRECTORY;
179            final String keyStoreFile = workingDir + File.separator + "system" + File.separator + "izou.keystore";
180            KeyStore keyStore = createKeyStore(keyStoreFile, "4b[X:+H4CS&avY<)");
181
182            KeyStore.PasswordProtection keyPassword = new KeyStore.PasswordProtection("Ev45j>eP}QTR?K9_".toCharArray());
183            KeyStore.Entry entry = keyStore.getEntry("izou_key", keyPassword);
184            key = ((KeyStore.SecretKeyEntry) entry).getSecretKey();
185        } catch(NullPointerException e) {
186            return null;
187        } catch (UnrecoverableEntryException | NoSuchAlgorithmException | KeyStoreException e) {
188            logger.error("Unable to retrieve key", e);
189        }
190
191        return key;
192    }
193
194    /**
195     * Stores the izou aes key in a keystore
196     *
197     * @param key the key to store
198     */
199    private void storeKey(SecretKey key) {
200        String workingDir = FileSystemManager.FULL_WORKING_DIRECTORY;
201        final String keyStoreFile = workingDir + File.separator + "system" + File.separator + "izou.keystore";
202        KeyStore keyStore = createKeyStore(keyStoreFile, "4b[X:+H4CS&avY<)");
203
204        try {
205            KeyStore.SecretKeyEntry keyStoreEntry = new KeyStore.SecretKeyEntry(key);
206            KeyStore.PasswordProtection keyPassword = new KeyStore.PasswordProtection("Ev45j>eP}QTR?K9_".toCharArray());
207            keyStore.setEntry("izou_key", keyStoreEntry, keyPassword);
208            keyStore.store(new FileOutputStream(keyStoreFile), "4b[X:+H4CS&avY<)".toCharArray());
209        } catch (NoSuchAlgorithmException | KeyStoreException
210                | CertificateException | IOException e) {
211            logger.error("Unable to store key", e);
212        }
213    }
214
215    /**
216     * Creates a new keystore for the izou aes key
217     *
218     * @param fileName the path to the keystore
219     * @param password the password to use with the keystore
220     * @return the newly created keystore
221     */
222    private KeyStore createKeyStore(String fileName, String password)  {
223        File file = new File(fileName);
224        KeyStore keyStore = null;
225        try {
226            keyStore = KeyStore.getInstance("JCEKS");
227            if (file.exists()) {
228                keyStore.load(new FileInputStream(file), password.toCharArray());
229            } else {
230                keyStore.load(null, null);
231                keyStore.store(new FileOutputStream(fileName), password.toCharArray());
232            }
233        } catch (CertificateException | IOException | KeyStoreException | NoSuchAlgorithmException e) {
234            logger.error("Unable to create key store", e);
235        }
236
237        return keyStore;
238    }
239}
